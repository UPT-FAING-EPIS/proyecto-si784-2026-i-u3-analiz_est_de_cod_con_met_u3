46@router.post("/register") 

47async def register( 

48 username: str = Form(...), 

49 password: str = Form(...), 

50 role: str = Form("developer"), 

51 db: Session = Depends(get_db), 

52) -> dict[str, Any]: 

53 """Registra un nuevo usuario en el sistema. 

54 

55 - username: Nombre de usuario único. 

56 - password: Contraseña en texto plano. 

57 - db: Sesión de base de datos inyectada. 

58 

59 Retorna los datos del usuario creado o un error si ya existe. 

60 """ 

61 repository = UserRepository(db) 

62 

63 # Verificar si el usuario ya existe 

64 existing_user = repository.get_user_by_username(username) 

65 if existing_user: 

66 raise HTTPException( 

67 status_code=status.HTTP_400_BAD_REQUEST, 

68 detail="El nombre de usuario ya existe", 

69 ) 

70 

71 try: 

72 user = repository.create_user(username=username, password=password, role=role) 

73 repository.log_action(user.id, "Registro de usuario") 

74 return { 

75 "id": user.id, 

76 "username": user.username, 

77 "role": user.role, 

78 "active_status": user.active_status, 

79 } 

80 except IntegrityError: 

81 db.rollback() 

82 raise HTTPException( 

83 status_code=status.HTTP_400_BAD_REQUEST, 

84 detail="Error al crear el usuario. El nombre de usuario podría estar duplicado.", 

85 )

88@router.post("/login") 

89async def login( 

90 username: str = Form(...), 

91 password: str = Form(...), 

92 db: Session = Depends(get_db), 

93) -> dict[str, Any]: 

94 repository = UserRepository(db) 

95 user = repository.get_user_by_username(username) 

96 

97 if not user: 

98 raise HTTPException( 

99 status_code=status.HTTP_401_UNAUTHORIZED, 

100 detail="Credenciales incorrectas", 

101 ) 

102 

103 is_valid_password = bcrypt.checkpw( 

104 password.encode("utf-8"), 

105 user.password_hash.encode("utf-8"), 

106 ) 

107 if not is_valid_password: 

108 raise HTTPException( 

109 status_code=status.HTTP_401_UNAUTHORIZED, 

110 detail="Credenciales incorrectas", 

111 ) 

112 

113 repository.update_login_status(user.id, True) 

114 repository.log_action(user.id, "Inicio de sesión") 

115 

116 return { 

117 "id": user.id, 

118 "username": user.username, 

119 "role": user.role, 

120 "is_logged_in": True, 

121 }

144@router.get("/callback/google") 

145async def callback_google(request: Request, db: Session = Depends(get_db)): 

146 try: 

147 token = await oauth.google.authorize_access_token(request) 

148 user_info = token.get('userinfo') 

149 if not user_info: 

150 raise HTTPException(status_code=400, detail="No user info returned from Google") 

151 

152 username = user_info.get('email', '').split('@')[0] or "google_user" 

153 

154 repository = UserRepository(db) 

155 user = repository.get_user_by_username(username) 

156 if not user: 

157 random_password = str(uuid.uuid4()) 

158 user = repository.create_user(username=username, password=random_password, role="developer") 

159 repository.log_action(user.id, "Registro de usuario via Google") 

160 

161 repository.update_login_status(user.id, True) 

162 repository.log_action(user.id, "Inicio de sesión via Google") 

163 

164 return RedirectResponse(url=f"/static/oauth_success.html?id={user.id}&username={user.username}&role={user.role}")
